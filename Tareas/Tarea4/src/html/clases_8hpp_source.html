199
200 for (int i = 0; i < filasMatriz; ++i) {

202 // Se genera un valor aleatorio dentro del rango adecuado
203 T valorAleatorio;
204
205 if constexpr (std::is_same_v<T, int>) {
206 valorAleatorio = distribucionEntera(generador);
207 }
208
209 else if constexpr (std::is_same_v<T, float>) {
210 valorAleatorio = distribucionFlotante(generador);
211 }
212 // Se asigna el valor aleatorio a la posicion correspondiente en la matriz
213 matrizResultado[i][j] = valorAleatorio;
214 }
215 }
216 }
217 else if constexpr (std::is_same_v<T, std::complex<float>>) {
218 // Si el tipo es complejo, se generan valores aleatorios para la parte real e imaginaria de cada numero complejo y se asignan a cada entrada de la matriz
219 std::uniform_real_distribution<float> distribucionReal(0.0, 100.0);
220 std::uniform_real_distribution<float> distribucionImaginaria(0.0, 100.0);
221
222 for (int i = 0; i < filasMatriz; ++i) {
223 for (int j = 0; j < columnasMatriz; ++j) {
224 // Se generan valores aleatorios para la parte real e imaginaria

227 // Se crea un numero complejo con los valores aleatorios y se asigna a la posicion correspondiente en la matriz

447 // Multiplicación para otros tipos de datos
